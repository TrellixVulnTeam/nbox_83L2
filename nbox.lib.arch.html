class InstallPython(StepOp)
def __init__(self, version: str = “3.9):

self.add_step(ShellCommand(f”chmod +x ./scripts/python{version}_install.sh”)) self.add_step(ShellCommand(f”./scripts/python{version}_install.sh”))

install_python = InstallPython() # init op install_python() # call it without defining the forward function
